基本思想#

downloadJsonIVR () {
  var data = {name: 'age'}
  data = JSON.stringify(data)
  data = new Blob([data])
  var a = document.createElement('a')
  var url = window.URL.createObjectURL(data)
  a.href = url
  a.download = 'what-you-want.json'
  a.click()
},
