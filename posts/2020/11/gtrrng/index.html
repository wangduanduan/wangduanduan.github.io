1. 将源码包上传到服务器, 并解压

安装依赖#

apt update
apt install autoconf \
libtool \
libtool-bin \
libjpeg-dev \
libsqlite3-dev \
libspeex-dev libspeexdsp-dev \
libldns-dev \
libedit-dev \
libtiff-dev \
libavformat-dev
libswscale-dev libsndfile-dev \
liblua5.1-0-dev
libcurl4-openssl-dev libpcre3-dev libopus-dev libpq-dev

配置#

./bootstrap.sh
./configure

make#

make && make install
