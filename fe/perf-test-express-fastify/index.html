机器信息：4C32G 测试工具：wrk Node: v14.17.0

express.js

'use strict'                                                                                                                                                                                  

const express = require('express')
const app = express()

app.get('/', function (req, res) {
  res.json({ hello: 'world' })
})

app.listen(3000)

fastify.js

'use strict'                                                                                                                                                                                  
const fastify = require('fastify')()

fastify.get('/', function (req, reply) {
  reply.send({ hello: 'world' })
})

fastify.listen(3000)
~
